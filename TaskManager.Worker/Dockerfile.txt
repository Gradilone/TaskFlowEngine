# Estágio de Compilação
FROM mcr.microsoft.com/dotnet/sdk:8.0 AS build
WORKDIR /src

# Copia os arquivos de projeto (.csproj) primeiro para cache de camadas
COPY ["TaskManager.Worker/TaskManager.Worker.csproj", "TaskManager.Worker/"]
# Se você tiver projetos de biblioteca (Models, Repositories), copie-os aqui:
COPY ["TaskManager.Models/TaskManager.Models.csproj", "TaskManager.Models/"]
COPY ["TaskManager.Repositories/TaskManager.Repositories.csproj", "TaskManager.Repositories/"]

RUN dotnet restore "TaskManager.Worker/TaskManager.Worker.csproj"

# Copia o restante e compila
COPY . .
WORKDIR "/src/TaskManager.Worker"
RUN dotnet build "TaskManager.Worker.csproj" -c Release -o /app/build

FROM build AS publish
RUN dotnet publish "TaskManager.Worker.csproj" -c Release -o /app/publish

# Estágio Final (Runtime)
FROM mcr.microsoft.com/dotnet/runtime:8.0 AS final
WORKDIR /app
COPY --from=publish /app/publish .
ENTRYPOINT ["dotnet", "TaskManager.Worker.dll"]
